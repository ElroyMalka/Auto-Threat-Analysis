    function escapeHTML(str) {

        if (!str) return "";

        return str.replace(/&/g, '&amp;').replace(/</g, '&lt;').replace(/>/g, '&gt;').replace(/"/g, '&quot;');

    }

    function calculateSeverity(likelihood, impact) {

        if (!likelihood || !impact) return { text: "טרם נקבע", class: "bg-none", inline: "" };

       

        const matrix = {

            "נמוכה": { "נמוכה": "נמוכה ביותר", "בינונית": "נמוכה", "גבוהה": "בינונית" },

            "בינונית": { "נמוכה": "נמוכה", "בינונית": "בינונית", "גבוהה": "גבוהה" },

            "גבוהה": { "נמוכה": "בינונית", "בינונית": "גבוהה", "גבוהה": "קריטית" }

        };

 

        const result = matrix[likelihood][impact];

        let colorClass = "bg-none";

        let inlineStyle = "";

       

        if (result === "נמוכה ביותר") { colorClass = "bg-very-low"; inlineStyle = "background-color:#92d050; color:#000;"; }

        else if (result === "נמוכה") { colorClass = "bg-low"; inlineStyle = "background-color:#00b050; color:#fff;"; }

        else if (result === "בינונית") { colorClass = "bg-medium"; inlineStyle = "background-color:#ffff00; color:#000;"; }

        else if (result === "גבוהה") { colorClass = "bg-high"; inlineStyle = "background-color:#ff0000; color:#fff;"; }

        else if (result === "קריטית") { colorClass = "bg-critical"; inlineStyle = "background-color:#c00000; color:#fff;"; }

 

        return { text: result, class: colorClass, inline: inlineStyle };

    }

    function importProject(event) {

        const file = event.target.files[0];

        if (!file) return;

        const reader = new FileReader();

        reader.onload = function(e) {

            try {

                const importedData = JSON.parse(e.target.result);

                if (importedData.data) {

                    projectData = importedData.data;

                    if (importedData.projectName) document.getElementById('projectNameInput').value = importedData.projectName;

                   

                    if (importedData.customReqs) {

                        let maxC = customReqCounter;

                        Object.keys(importedData.customReqs).forEach(id => {

                            reqDB[id] = importedData.customReqs[id];

                            let num = parseInt(id.substring(1));

                            if (!isNaN(num) && num >= maxC) maxC = num + 1;

                        });

                        customReqCounter = maxC;

                    }

 

                    alert("✅ פרויקט סייבר נטען בהצלחה!");

                    if (!document.getElementById('step2').classList.contains('hidden')) {

                        document.getElementById('displayProjectName').innerText = `פרויקט: ${document.getElementById('projectNameInput').value}`;

                        loadThreats();

                    } else if (importedData.projectName) goToStep2();

                } else alert("❌ קובץ שמירה לא תקין.");

            } catch (err) { alert("❌ שגיאה בקריאת הקובץ."); }

            document.getElementById('importFile').value = '';

        };

        reader.readAsText(file);

    }
